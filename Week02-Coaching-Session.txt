async function getStudentCourses(studentId) {
  const query = `
    SELECT students.name, courses.title, enrollments.grade
    FROM enrollments
    JOIN students ON enrollments.student_id = students.id
    JOIN courses ON enrollments.course_id = courses.id
    WHERE students.id = $1;
  `;
  const result = await pool.query(query, [studentId]);
  return result.rows;
}

I use a connection pool from `pg` and export this function. The server calls it, 
waits for the rows, and then passes the data to the route handler.

app.get('/students/:id/courses', async (req, res) => {
  const studentId = req.params.id;
  const courses = await getStudentCourses(studentId);
  res.render('student_courses', { student: courses[0], courses });
});

This route maps `/students/:id/courses` to a handler that fetches the student's 
courses and renders the EJS view. The `:id` is dynamic, so it works for any student.
